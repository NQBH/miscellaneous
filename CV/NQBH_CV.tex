\section{Master Thesis}

\cvitem{Title}{\emph{Staggered and well-balanced discretization of shallow-water equations} [\href{https://github.com/NQBH/reference/blob/master/NQBH_Master_Thesis/NQBH_Master_Thesis.pdf}{PDF}]}

\cvitem{Supervisor}{Prof. \href{http://seguin.perso.math.cnrs.fr/}{Nicolas Seguin}}

\cvitem{Description}{This thesis investigates some staggered well-balanced finite volume schemes for the shallow water equations. A new staggered scheme, called ``offset equilibrium'', is developed to conquer the \textit{dry-wet transition} issues where almost all the traditional finite volume schemes collapse. Higher-order extensions for this staggered scheme are also mentioned.}
